Private Sub Text1_Change()
Dim n As Integer
n = Val(Text1.Text)
Select Case n
Case 1
Label2 = " Janeiro"
Case 2
Label2 = " Fevereiro"
Case 3
Label2 = " Março"
Case 4
Label2 = " Abril"
Case 5
Label2 = " Maio"
Case 6
Label2 = " Junho"
Case 7
Label2 = " Julho"
Case 8
Label2 = " Agosto"
Case 9
Label2 = " Setembro"
Case 10
Label2 = " Outubro"
Case 11
Label2 = " Novembro"
Case 12
Label2 = " Dezembro"
Case Else
Label2 = ""
End Select

End Sub
